Reproducible base environment for Android development

üìãOverview

This bootstrap system provides a reproducible, minimal Termux setup with proot-distro as the foundation for running full Linux environments on Android. The system is designed to be:

  • ‚úÖ Idempotent - Safe to run multiple times
  • ‚úÖ Minimal - Only essential packages
  • ‚úÖ Versioned - Tracks configuration state
  • ‚úÖ Automated - Single command setup

‚ö°Quick Start

Bootstrap your Termux environment with a single command:

curl -L https://your-domain.com/termux-bootstrap.sh | bash

Or if hosting on GitHub:

pkg install curl && \
curl -L https://raw.githubusercontent.com/YOUR-USERNAME/termux-setup/main/bootstrap.sh | bash

üì¶Main Bootstrap Script

bootstrap.sh
#!/data/data/com.termux/files/usr/bin/bash
# Termux Bootstrap Script v1.0
# Reproducible base environment setup for Android

set -euo pipefail

# Configuration
BOOTSTRAP_VERSION="1.0.0"
BOOTSTRAP_DIR="$HOME/.termux-bootstrap"
CONFIG_FILE="$BOOTSTRAP_DIR/config.json"
LOG_FILE="$BOOTSTRAP_DIR/bootstrap.log"

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Logging functions
log() {
    echo -e "${GREEN}[$(date '+%Y-%m-%d %H:%M:%S')]${NC} $1" | tee -a "$LOG_FILE"
}

error() {
    echo -e "${RED}[ERROR]${NC} $1" | tee -a "$LOG_FILE"
    exit 1
}

warn() {
    echo -e "${YELLOW}[WARN]${NC} $1" | tee -a "$LOG_FILE"
}

info() {
    echo -e "${BLUE}[INFO]${NC} $1" | tee -a "$LOG_FILE"
}

# Create bootstrap directory
init_bootstrap() {
    if [[ ! -d "$BOOTSTRAP_DIR" ]]; then
        mkdir -p "$BOOTSTRAP_DIR"
        log "Created bootstrap directory: $BOOTSTRAP_DIR"
    fi
}

# Check if already bootstrapped
check_existing() {
    if [[ -f "$CONFIG_FILE" ]]; then
        existing_version=$(grep -oP '"version":\s*"\K[^"]+' "$CONFIG_FILE" 2>/dev/null || echo "unknown")
        warn "System already bootstrapped (version: $existing_version)"
        read -p "Do you want to re-run bootstrap? (y/N): " -n 1 -r
        echo
        if [[ ! $REPLY =~ ^[Yy]$ ]]; then
            log "Bootstrap cancelled by user"
            exit 0
        fi
    fi
}

# Setup storage permissions
setup_storage() {
    if [[ ! -d "$HOME/storage" ]]; then
        log "Setting up storage access..."
        termux-setup-storage
        sleep 3  # Wait for permission dialog
        
        # Wait for user to grant permission
        while [[ ! -d "$HOME/storage" ]]; do
            warn "Please grant storage permission in the Android dialog"
            sleep 2
        done
        log "Storage access configured"
    else
        log "Storage access already configured"
    fi
}

# Update and upgrade base packages
update_system() {
    log "Updating package repositories..."
    pkg update -y || error "Failed to update packages"
    
    log "Upgrading installed packages..."
    pkg upgrade -y || error "Failed to upgrade packages"
}

# Install essential packages
install_packages() {
    ESSENTIAL_PACKAGES=(
        # Core utilities
        "proot"
        "proot-distro"
        "curl"
        "wget"
        "git"
        "openssh"
        
        # Terminal essentials
        "termux-api"
        "termux-tools"
        
        # Text editors (minimal)
        "nano"
        "vim"
        
        # Build essentials (for later compilation needs)
        "build-essential"
        "pkg-config"
        
        # Archive tools
        "tar"
        "gzip"
        "zip"
        "unzip"
        
        # Process management
        "htop"
        "tree"
        
        # Network tools
        "net-tools"
        "nmap"
    )
    
    log "Installing essential packages..."
    for package in "${ESSENTIAL_PACKAGES[@]}"; do
        info "Installing $package..."
        pkg install -y "$package" || warn "Failed to install $package"
    done
}

# Configure shell environment
setup_shell() {
    log "Configuring shell environment..."
    
    # Create .bashrc if it doesn't exist
    if [[ ! -f "$HOME/.bashrc" ]]; then
        cat > "$HOME/.bashrc" << 'EOF'
# Termux Bootstrap Shell Configuration
export TERM=xterm-256color
export LANG=en_US.UTF-8
export LC_ALL=en_US.UTF-8

# Aliases
alias ll='ls -alF'
alias la='ls -A'
alias l='ls -CF'
alias pd='proot-distro'
alias pda='proot-distro login alpine --shared-tmp'
alias pdd='proot-distro login debian --shared-tmp'
alias pdu='proot-distro login ubuntu --shared-tmp'

# Functions
pdrun() {
    if [ -z "$1" ] || [ -z "$2" ]; then
        echo "Usage: pdrun  "
        return 1
    fi
    proot-distro login "$1" --shared-tmp -- "$2"
}

# Custom prompt
PS1='\[\033[01;32m\]termux\[\033[00m\]:\[\033[01;34m\]\w\[\033[00m\]\$ '

# Bootstrap version info
export TERMUX_BOOTSTRAP_VERSION="1.0.0"
EOF
        log "Created .bashrc configuration"
    else
        log ".bashrc already exists, skipping"
    fi
    
    # Source the bashrc
    source "$HOME/.bashrc"
}

# Setup proot-distro
setup_proot_distro() {
    log "Configuring proot-distro..."
    
    # Create custom proot-distro configuration
    PROOT_CONFIG_DIR="$PREFIX/etc/proot-distro"
    
    if [[ ! -d "$PROOT_CONFIG_DIR" ]]; then
        mkdir -p "$PROOT_CONFIG_DIR"
    fi
    
    # Install Alpine as the default minimal distro
    log "Installing Alpine Linux (minimal base)..."
    proot-distro install alpine || warn "Alpine may already be installed"
    
    # Create a launch script for quick access
    cat > "$HOME/enter-alpine.sh" << 'EOF'
#!/data/data/com.termux/files/usr/bin/bash
# Quick entry to Alpine with shared tmp
exec proot-distro login alpine --shared-tmp
EOF
    chmod +x "$HOME/enter-alpine.sh"
    log "Created quick entry script: ~/enter-alpine.sh"
}

# Create bootstrap configuration
save_config() {
    log "Saving bootstrap configuration..."
    
    cat > "$CONFIG_FILE" << EOF
{
    "version": "$BOOTSTRAP_VERSION",
    "timestamp": "$(date -Iseconds)",
    "packages": [
$(printf '        "%s",\n' "${ESSENTIAL_PACKAGES[@]}" | sed '$ s/,$//')
    ],
    "distros": ["alpine"],
    "android_version": "$(getprop ro.build.version.release)",
    "device": "$(getprop ro.product.model)",
    "arch": "$(uname -m)"
}
EOF
    log "Configuration saved to $CONFIG_FILE"
}

# Verification step
verify_installation() {
    log "Verifying installation..."
    
    VERIFY_COMMANDS=(
        "proot --version"
        "proot-distro list"
        "git --version"
        "curl --version | head -n1"
    )
    
    failed=0
    for cmd in "${VERIFY_COMMANDS[@]}"; do
        if eval "$cmd" &>/dev/null; then
            info "‚úì $cmd"
        else
            error "‚úó $cmd failed"
            ((failed++))
        fi
    done
    
    if [[ $failed -eq 0 ]]; then
        log "All verification checks passed!"
    else
        warn "$failed verification checks failed"
    fi
}

# Main execution
main() {
    echo -e "${GREEN}‚ïî‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïó${NC}"
    echo -e "${GREEN}‚ïë   Termux Bootstrap System v1.0    ‚ïë${NC}"
    echo -e "${GREEN}‚ïö‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïù${NC}"
    echo
    
    init_bootstrap
    check_existing
    
    log "Starting bootstrap process..."

    update_system
    install_packages
    setup_shell
    setup_proot_distro

    verify_installation
    
    echo
    echo -e "${GREEN}‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê${NC}"
    log "Bootstrap completed successfully!"
    echo -e "${GREEN}‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê${NC}"

main "$@"

#!/data/data/com.termux/files/usr/bin/bash
# Backup Termux configuration

BACKUP_DIR="$HOME/storage/shared/termux-backups"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="termux-backup-$TIMESTAMP.tar.gz"

mkdir -p "$BACKUP_DIR"

echo "Creating backup..."
tar -czf "$BACKUP_DIR/$BACKUP_FILE" \
    --exclude="$HOME/storage" \
    --exclude="$HOME/.cache" \
    --exclude="*.log" \
    "$HOME/.termux" \
    "$HOME/.bashrc" \
    "$HOME/.termux-bootstrap" \
    "$PREFIX/etc/proot-distro" \
    2>/dev/null

echo "Backup saved to: $BACKUP_DIR/$BACKUP_FILE"

# Save installed package list
pkg list-installed > "$BACKUP_DIR/packages-$TIMESTAMP.txt"
proot-distro list > "$BACKUP_DIR/distros-$TIMESTAMP.txt"

echo "Package lists saved"
restore-termux.sh
#!/data/data/com.termux/files/usr/bin/bash
# Restore Termux configuration from backup

BACKUP_DIR="$HOME/storage/shared/termux-backups"

if [[ ! -d "$BACKUP_DIR" ]]; then
    echo "No backup directory found"
    exit 1
fi

# List available backups
echo "Available backups:"
ls -1 "$BACKUP_DIR"/*.tar.gz 2>/dev/null | nl

read -p "Enter backup number to restore: " choice
BACKUP_FILE=$(ls -1 "$BACKUP_DIR"/*.tar.gz 2>/dev/null | sed -n "${choice}p")

if [[ -z "$BACKUP_FILE" ]]; then
    echo "Invalid selection"
    exit 1
fi

echo "Restoring from: $BACKUP_FILE"
tar -xzf "$BACKUP_FILE" -C / 2>/dev/null

echo "Restoration complete"
echo "Please restart Termux"
update-bootstrap.sh
#!/data/data/com.termux/files/usr/bin/bash
# Update bootstrap components

set -euo pipefail

BOOTSTRAP_DIR="$HOME/.termux-bootstrap"
CONFIG_FILE="$BOOTSTRAP_DIR/config.json"

echo "Updating Termux packages..."
pkg update -y
pkg upgrade -y

echo "Updating proot-distro..."
pkg install -y proot-distro

echo "Updating installed distributions..."
for distro in $(proot-distro list | grep "Installed" | awk '{print $1}'); do
    echo "Updating $distro..."
    proot-distro login "$distro" -- sh -c "
        if command -v apk >/dev/null; then
            apk update && apk upgrade
        elif command -v apt >/dev/null; then
            apt update && apt upgrade -y
        fi
    "
done

# Update timestamp in config
if [[ -f "$CONFIG_FILE" ]]; then
    tmp=$(mktemp)
    jq '.last_update = now | todate' "$CONFIG_FILE" > "$tmp" && mv "$tmp" "$CONFIG_FILE"
fi

echo "Update complete!"

üéØUsage Patterns

1Fresh Install

Run the bootstrap script on a new Termux installation to set up your base environment.

2Daily Use

Use pda alias to quickly enter Alpine with shared tmp for file access.

3Backup

Regular backups to Android storage ensure you can recover your setup.

4Updates

Run update script periodically to keep all components current.

üö¶Next Steps

Once your base Termux environment is set up:

  1. Install NixOS in proot-distro (if available) or use Alpine/Debian as base
  2. Install Nix package manager in your chosen distribution
  3. Set up home-manager with your existing configurations
  4. Sync your dotfiles using your preferred method
Ready for Phase 2: With this reproducible base, you can now build your Nix/home-manager environment on top, knowing the foundation is consistent across devices.

Important Notes:
  • Always run termux-setup-storage before accessing Android files
  • The --shared-tmp flag is crucial for file sharing between Termux and proot
  • Some packages may require additional configuration for optimal performance
  • Keep backups before major updates or experiments
