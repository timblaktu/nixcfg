110impl SyntaxKind {
111    /// Returns true if this token is a literal, such as an integer or a string
112    pub fn is_literal(self) -> bool {
113        matches!(self, TOKEN_FLOAT | TOKEN_INTEGER | TOKEN_PATH | TOKEN_URI)
114    }
115
116    /// Returns true if this token should be used as a function argument.
117    /// ```ignore
118    /// Example:
119    /// add 1 2 + 3
120    /// ^   ^ ^ ^
121    /// |   | | +- false
122    /// |   | +- true
123    /// |   +- true
124    /// +- true
125    /// ```
126    pub fn is_fn_arg(self) -> bool {
127        match self {
128            TOKEN_REC | TOKEN_L_BRACE | TOKEN_L_BRACK | TOKEN_L_PAREN | TOKEN_STRING_START
129            | TOKEN_IDENT => true,
130            _ => self.is_literal(),
131        }
132    }
133    /// Returns true if this token is a comment, whitespace, or similar, and
134    /// should be skipped over by the parser.
135    pub fn is_trivia(self) -> bool {
136        matches!(self, TOKEN_COMMENT | TOKEN_WHITESPACE)
137    }
